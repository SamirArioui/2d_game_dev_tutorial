# Exercise 04 — solution
cmake_minimum_required(VERSION 3.16)
project(ex04 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

include(FetchContent)
FetchContent_Declare(
    fmt
    GIT_REPOSITORY https://github.com/fmtlib/fmt
    GIT_TAG 11.0.2      # pin the version; don't track a moving branch
    GIT_SHALLOW TRUE    # skip the full history — faster clone
)
FetchContent_MakeAvailable(fmt)  # downloads on first configure, then caches in build/

add_executable(app main.cpp)
target_link_libraries(app PRIVATE fmt::fmt)  # the target fmt exposes for consumers
