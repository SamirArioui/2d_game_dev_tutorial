# Exercise 03 — solution
cmake_minimum_required(VERSION 3.16)
project(ex03 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_library(mathlib src/mathlib.cpp)

# PUBLIC: needed to build mathlib itself, and re-exported to anything linking it.
target_include_directories(mathlib PUBLIC include)

add_executable(app app/main.cpp)
target_link_libraries(app PRIVATE mathlib)  # app inherits the include/ path from mathlib
